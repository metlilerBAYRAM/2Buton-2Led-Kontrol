void setup(){
  pinMode(4,INPUT_PULLUP);
  pinMode(5,INPUT_PULLUP);
  pinMode(8,OUTPUT);
  pinMode(9,OUTPUT);
}
void loop(){
   
   if(digitalRead(4)==0)digitalWrite(8,1);else digitalWrite(8,0);
   if(digitalRead(5)==0)digitalWrite(9,1);else digitalWrite(9,0);
}
